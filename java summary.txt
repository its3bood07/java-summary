            <span class="lang-label ar-label">عربي</span>

            <pre><code><span class="kw">int</span> x=10;

          </div>
          <div class="lang-box" dir="ltr">
            <span class="lang-label en-label">English</span>
            <ul class="points">
              <li><strong>Prefix (++x):</strong> increment first, then use</li>
              <li><strong>Postfix (x++):</strong> use first, then increment</li>
            </ul>
          </div>
        </div>
      </div>

          <tr><td><code>a *= b</code></td><td><code>a = a * b</code></td></tr>
          <tr><td><code>a /= b</code></td><td><code>a = a / b</code></td></tr>
          <tr><td><code>a %= b</code></td><td><code>a = a % b</code></td></tr>

          <div class="lang-box">
            <span class="lang-label ar-label">عربي</span>
